```bash
npm install --save react-fitmeplz
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-fitmeplz'
import 'react-fitmeplz/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
